def ret_weekday_corr(symbol,period_start,period_end):
    ret = returns(assets=symbols(symbol),start=period_start,end=period_end,)
#    ret = ret.dropna()
    print("symbol is: ",symbol)
    ret_1_l = []
    ret_2_l = []
    ret_3_l = []
    ret_4_l = []
    ret_5_l = []

    for i in range(len(ret)):
        if(ret.index[i].date().weekday()==0):
            ret_1_l.append(ret[i])
        elif(ret.index[i].date().weekday()==1):
            ret_2_l.append(ret[i])
        elif(ret.index[i].date().weekday()==2):
            ret_3_l.append(ret[i])
        elif(ret.index[i].date().weekday()==3):
            ret_4_l.append(ret[i])
        elif(ret.index[i].date().weekday()==4):
            ret_5_l.append(ret[i])
    
    N = min(len(ret_1_l),len(ret_2_l),len(ret_3_l),len(ret_4_l),len(ret_5_l))
#    print("N is: ",N)
    ret_1 = pd.DataFrame(ret_1_l[:N])
    ret_2 = pd.DataFrame(ret_2_l[:N])
    ret_3 = pd.DataFrame(ret_3_l[:N])
    ret_4 = pd.DataFrame(ret_4_l[:N])
    ret_5 = pd.DataFrame(ret_5_l[:N])

    ret_ll = [ret_1_l[:N],ret_2_l[:N],ret_3_l[:N],ret_4_l[:N],ret_5_l[:N]]
    ret_df = pd.DataFrame(ret_ll)
    return [ret_df,N]

corr_max = []
for symbol in symbol_list[:L]:
    corr_sign.append([])
    [ret_df,N] = ret_weekday_corr(symbol,period_start,period_end)
    ret_corr = ret_df.T.corr().values
    #t = 1.648 # t-statistic is 0.95 for N-2 dof
    #t = 2.059 # t-statistic is 0.98 for N-2 dof
    t = 2.586 # t-statistic is 0.995 for N-2 dof
    #t = 3.107 # t-statistic is 0.998 for N-2 dof
    r = t/np.sqrt(N-2+(t**2))
    flag = 0
    for i in range(5):
        for j in range(5):
            if(abs(ret_corr[i][j])>=r and i!=j):
                corr_sign[-1].append([i,j])
                corr_hist[i,j] += 1
                flag = 1

    if(flag==1):
        ret_corr[ret_corr==1.0]=0.0
        corrmax_ind = np.unravel_index(np.argmax(ret_corr, axis=None), ret_corr.shape)
        corrmax = ret_corr.max()
        corr_max.append([corrmax,corrmax_ind,symbol])
    
    flag = 0
